    清空某个文件 # 

    echo -n '' > /logs/test.log
# rm 大文件可能导致磁盘占用不释放
# echo '' > /logs/test.log 删除不掉时, 加 -n 参数尝试

    宿主机上以root进入容器 # 

    docker exec --privileged -ti <容器ID> bash

    atop #

    查看指定容器的进程 # 

    atop
J
# 输入CID

    查看指定进程ID # 

    atop
I
# 输入PID

    读取历史日志 # 

    atop -r /path/to/file
# t键向前翻页，T键向后翻页
# b键跳转到指定时间，时间格式为hh:mm
